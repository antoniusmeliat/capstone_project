Hybrid Query Selection Workflow for Bayesian Optimization¶

This notebook demonstrates a hybrid strategy for selecting query points in a Bayesian Optimization setting.
The approach combines:

  • Acquisition functions (Expected Improvement and Upper Confidence Bound) for systematic exploration/exploitation.
  • Targeted perturbations around top-performing points for flexibility and faster convergence.

Why Hybrid?¶

  • Pure exploration can miss subtle nonlinear interactions.
  • Pure exploitation risks premature convergence.
  • A hybrid approach accelerates learning while maintaining diversity.

Directions for Use:¶

  1. Run the setup cells to load functions for GP fitting, acquisition functions, perturbations, and visualisations.
  2. Provide your observed data (X, Y) from previous rounds.
  3. Fit the GP model and compute acquisition scores.
  4. Generate candidate queries using the hybrid strategy.
  5. Visualise results to guide your next round of submissions.

def fit_gp(
    X: np.ndarray,
    Y: np.ndarray,
    *,
    random_state: int = 0,
    # GP fitting knobs (auto-scaled if None)
    n_restarts: Optional[int] = None,
    nu: float = 2.5,
    standardize_y: bool = True,
    suppress_convergence_warnings: bool = False,
    length_scale_lower: Optional[float] = None,
    length_scale_upper: Optional[float] = None,
    noise_lower: Optional[float] = None,
    noise_upper: Optional[float] = None,
    # Proposal knobs (auto-scaled by dimension if None)
    proposal_k_final: int = 5,
    proposal_random_state: Optional[int] = None,
    proposal_n_global: Optional[int] = None,
    proposal_n_trust: Optional[int] = None,
    proposal_trust_radius: Optional[float] = None,
    proposal_kappa: float = 2.0,
) -> Tuple[GaussianProcessRegressor, Tuple[List[np.ndarray], List[str]]]:
    """Fit an ARD Matern+White GP with dimension/data-aware bounds and propose queries.
    Returns: gp, (numeric_candidates, formatted_candidates)
    """

    # --- 0) Dimension-aware config ---
    if suppress_convergence_warnings:
        warnings.filterwarnings("ignore", category=ConvergenceWarning)

    if X.ndim == 1:
        X = X.reshape(-1, 1)
    n, D = int(X.shape[0]), int(X.shape[1])

    # Restarts: scale by D and n, cap at 60
    if n_restarts is None:
        import math
        n_restarts = int(20 + 5 * min(D, 8) + 5 * math.log10(max(n, 10)))
        n_restarts = min(max(n_restarts, 20), 60)

    # Y standardization & normalize_y flag
    if standardize_y:
        scaler = StandardScaler(with_mean=True, with_std=True)
        Y_fit = scaler.fit_transform(Y.reshape(-1, 1)).ravel()
        normalize_y = False
    else:
        Y_fit = Y
        normalize_y = True


    # Per-dimension spread and geometric mean (more robust in heterogeneous D)
    s_per_dim = np.std(X, axis=0) + 1e-12
    gm_s = float(np.exp(np.mean(np.log(s_per_dim))))
    

    # If caller provides explicit bounds, keep them; else compute
    if length_scale_lower is None:
        # Lower bound shrinks with sqrt(D) to keep local sensitivity in higher-D
        length_scale_lower = max(1e-3, 0.25 * gm_s / max(np.sqrt(D), 1.0))

    if length_scale_upper is None:
        # Upper bound grows with sqrt(D) to allow smoother functions in higher-D
        # Use a generous starting cap; we will widen adaptively if we still peg.
        length_scale_upper = min(30.0, 6.0 * gm_s * max(np.sqrt(D), 1.0))


    # Noise bounds: tighter for standardized Y
    if noise_lower is None:
        noise_lower = 1e-12 if standardize_y else 1e-14
    if noise_upper is None:
        noise_upper = 0.75 if standardize_y else max(1e-3, 0.5 * float(np.std(Y)))

#=====
    # Proposal defaults by dimension
    if proposal_random_state is None:
        proposal_random_state = random_state
    if proposal_n_global is None:
        proposal_n_global = 2000 * min(D, 8)
    if proposal_n_trust is None:
        proposal_n_trust = 16 + 8 * min(D, 8)
    if proposal_trust_radius is None:
        proposal_trust_radius = 0.03 / max(np.sqrt(D), 1.0)

    # --- 2) Kernel builder (ARD Matern + White) ---
    
    # Heuristic ARD init: start each dimension at its own spread# Heuristic ARD init: start each dimension at
    c_init = 1.0                # neutral; tune to 0.5–1.5 if needed
    ls0 = c_init * (np.std(X, axis=0) + 1e-12)

    def _make_kernel(ls_up: float, ls_init: np.ndarray):
        amp_bounds = (1e-6, 1e3) if standardize_y else (1e-6, 1e4)
        return (
            ConstantKernel(1.0, amp_bounds)
            * Matern(length_scale=ls_init,
                     length_scale_bounds=(length_scale_lower, ls_up),
                     nu=nu)
                   + WhiteKernel(noise_level=1e-4,
                          noise_level_bounds=(noise_lower, noise_upper),))


    # --- 3) Fit with bounded retry (at most 1 widening, capped upper bound) ---
    gp = None
    ls_up = float(length_scale_upper)
    tries, max_tries = 0, 6
    while tries <= max_tries:
        kernel = _make_kernel(ls_up,ls0)
        gp = GaussianProcessRegressor(
            kernel=kernel,
            normalize_y=normalize_y,
            n_restarts_optimizer=max(n_restarts,30), #ensure at least 30 restarts
            random_state=random_state,
        )
        gp.fit(X, Y_fit)


        # Inspect learned params and see if we hit upper bounds
        need_retry = False
        ls_hit_upper = False  # local helper to decide if we should smooth ls

        
        try:
            # Kernel layout: Constant * (Matern + White)
            matern = gp.kernel_.k1.k2
            white = gp.kernel_.k2


            # Length-scale upper bound peg check (ARD vector)
            ls_learned     = np.asarray(matern.length_scale, dtype=float)
            ls_upper_bound = float(matern.length_scale_bounds[1])
            ls_hit_upper   = np.any(ls_learned >= 0.985 * ls_upper_bound)

            # Noise upper/lower bound checks
            noise_learned     = float(white.noise_level)
            noise_lower_bound = float(white.noise_level_bounds[0])
            noise_upper_bound = float(white.noise_level_bounds[1])
        
            # Retry if either upper bound is effectively hit
            need_retry = ls_hit_upper or (noise_learned >= 0.985 * noise_upper_bound)

            # If noise hugs the lower bound, relax it further next time (avoid fake smoothness)
            if noise_learned <= 1.02 * noise_lower_bound:  
                noise_lower = max(1e-14, noise_lower * 0.1)

            # If noise hugs the upper bound, lift the cap (preserving your variable)
            if noise_learned >= 0.985 * noise_upper_b:
                noise_upper = min(
                    (noise_upper if noise_upper is not None else noise_upper_bound) * 2.0,
                    2.0 if standardize_y else max(1.0, 10.0 * float(np.std(Y)))
                )
                
        except Exception:
            # If we cannot read kernel internals, assume no retry is needed
            need_retry = False
            ls_hit_upper = False
    
           # Exit when a retry is not needed (good fit achieved)
        if not need_retry:
            break


        # Prepare next try: widen and smooth, dimension-aware using your 'D
        tries += 1
         # Widen factor stronger than 3×; ceiling grows with sqrt(D)
        ls_up = min(ls_up * 5.0, 1500.0* max(np.sqrt(D), 1.0))  # D-aware cap

        # If length-scale pegged, start closer to the smoother regime
        if ls_hit_upper:
                ls0 = np.clip(ls0 * 2.0, length_scale_lower, ls_up)

        # Increase restarts modestly as we retry (uses your n_restart
        n_restarts = max(n_restarts + 8, 100) 

    
    # --- 4) Proposal step (EI + UCB + trust-region) ---
    rng = np.random.default_rng(proposal_random_state)

    def _lhs(n_samples: int, d: int, rng_):
        cut = np.linspace(0, 1, n_samples + 1)
        a, b = cut[:-1], cut[1:]
        Xg = rng_.uniform(a, b, size=(d, n_samples)).T
        for j in range(d):
            rng_.shuffle(Xg[:, j])
        return Xg

    
    Xg = _lhs(proposal_n_global, D, rng)
    Y_best_val = float(np.max(Y_fit))
    X_best = X[int(np.argmax(Y_fit))]
    deltas = rng.uniform(-proposal_trust_radius, proposal_trust_radius, size=(proposal_n_trust, D))
    Xt = np.clip(X_best + deltas, 0.0, 1.0 - 1e-6)

    Xc = np.vstack([Xg, Xt])

    mean, std = gp.predict(Xc, return_std=True)
    std = np.maximum(std, 1e-9)
    xi = 0.01
    imp = mean - Y_best_val - xi
    Z = imp / std
    ei = imp * norm.cdf(Z) + std * norm.pdf(Z)
    tight = std <= 1e-9
    if np.any(tight):
        ei[tight] = np.maximum(0.0, imp[tight])

    ucb = mean + proposal_kappa * std
    order = np.lexsort((-ucb, -ei))
    Xc_sorted = Xc[order]

    def _fmt_portal(x: np.ndarray) -> str:
        x = np.clip(np.asarray(x, float), 0.0, 1.0 - 1e-6)
        return "-".join(f"{v:.6f}" for v in x)

    seen: set = set()
    numeric: List[np.ndarray] = []
    formatted: List[str] = []

    for x in Xc_sorted:
        fx = _fmt_portal(x)
        if fx not in seen:
            seen.add(fx)
            numeric.append(x)
            formatted.append(fx)
            if len(numeric) >= proposal_k_final:
                break

    return gp, (numeric, formatted)

def expected_improvement(Xc, gp, Y_best, xi=0.01, eps=1e-9):
    # Numerically stable EI (protect against std→0)
    mean, std = gp.predict(Xc, return_std=True)
    std = np.maximum(std, eps)
    imp = mean - Y_best - xi
    Z = imp / std
    ei = imp * norm.cdf(Z) + std * norm.pdf(Z)
    # For std≈0: EI should be ~max(0, imp)
    ei[std <= eps] = np.maximum(0.0, imp[std <= eps])
    return ei


def upper_confidence_bound(Xc, gp, kappa=2.0):
    mean, std = gp.predict(Xc, return_std=True)
    return mean + kappa * std

def generate_perturbations(top_points, perturbation_rate=0.05, bounds=(0, 1), rng=None):
    rng = np.random.default_rng(0) if rng is None else rng
    perturbed_points = []
    for point in top_points:
        delta = rng.uniform(-perturbation_rate, perturbation_rate, size=point.shape)
        p = np.clip(point + delta, bounds[0], bounds[1])
        perturbed_points.append(p)
    return np.array(perturbed_points)

def hybrid_query_selection(X_grid, gp, Y_best, top_points, n_queries=10):
    ei_scores = expected_improvement(X_grid, gp, Y_best)
    ucb_scores = upper_confidence_bound(X_grid, gp)
    ei_top_idx = np.argsort(ei_scores)[-n_queries//2:]
    ucb_top_idx = np.argsort(ucb_scores)[-n_queries//2:]
    acquisition_candidates = X_grid[np.unique(np.concatenate([ei_top_idx, ucb_top_idx]))]
    perturbation_candidates = generate_perturbations(top_points, perturbation_rate=0.05)
    combined_candidates = np.vstack([acquisition_candidates, perturbation_candidates])
    return combined_candidates

def plot_gp_1d(X, Y, gp, dim=0, n_points=200, beta=1.96):
    """
    1D slice: vary `dim` ∈ [0,1], fix others at median.
    Clean fill_between and no undefined variables.
    """
    d = X.shape[1]
    fixed = np.median(X, axis=0)
    grid = np.linspace(0.0, 1.0, n_points)
    X_full = np.tile(fixed, (n_points, 1))
    X_full[:, dim] = grid

    mean, std = gp.predict(X_full, return_std=True)

    fig, ax = plt.subplots(figsize=(10, 5))
    ax.plot(grid, mean, color='C0', label='GP mean')
    ax.fill_between(grid, mean - beta * std, mean + beta * std, color='C0', alpha=0.2, label='95% CI')

    # Observed points projected on this dimension
    ax.scatter(X[:, dim], Y, color='C3', s=20, alpha=0.8, label='Observed')

    ax.set_xlabel(f'Dimension {dim+1}')
    ax.set_ylabel('Output')
    ax.set_title(f'1D GP slice (vary dim {dim+1}, others fixed at median)')
    ax.legend(loc='best')
    plt.show()

def plot_all_1d_slices(X, Y, gp, dims=None, n_points=200, beta=1.96, 
                       show_ei=True, show_ucb=True, Y_best=None):

    For each dimension, vary that dim on [0,1] and fix others at the median.
    Plot GP posterior mean + 95% CI; overlay EI/UCB; mark EI slice maximum.
    """

        dims = list(range(X.shape[1]))
    fixed = np.median(X, axis=0)


    # Ensure best value is available on the GP training scale

        ucb = upper_confidence_bound(X_full, gp, kappa=2.5) if show_ucb else None

        ax1.plot(grid, mean, color='blue', label='GP mean')

def plot_2d_ei_slice(X, Y, gp, d1, d2, n_points=100, Y_best=None):
    """
    Vary two dimensions (d1, d2); fix others at median.
    Show GP mean contours and EI contours; mark EI slice maximum.
    """

    # Use GP training scale for Y_best by default
    if Y_best is None:
        Y_best = float(np.max(gp.y_train_))

    fixed = np.median(X, axis=0)
    g1 = np.linspace(0, 1, n_points)
    g2 = np.linspace(0, 1, n_points)
    G1, G2 = np.meshgrid(g1, g2)

    X_full = np.tile(fixed, (n_points*n_points, 1))
    X_full[:, d1] = G1.ravel()
    X_full[:, d2] = G2.ravel()

    mean, std = gp.predict(X_full, return_std=True)
    ei = expected_improvement(X_full, gp, Y_best=Y_best)

    mean_2d = mean.reshape(n_points, n_points)
    ei_2d = ei.reshape(n_points, n_points)

    # EI max index on grid
    max_idx = np.unravel_index(np.argmax(ei_2d), ei_2d.shape)
    max_x1 = g1[max_idx[1]]  # note meshgrid col/row indices
    max_x2 = g2[max_idx[0]]

    fig, ax = plt.subplots(1, 2, figsize=(14, 5))

    c0 = ax[0].contourf(G1, G2, mean_2d, cmap='viridis')
    fig.colorbar(c0, ax=ax[0])
    ax[0].scatter(X[:, d1], X[:, d2], color='red', s=20, label='Observed')
    ax[0].set_title(f"GP mean (dims {d1+1} vs {d2+1})")
    ax[0].set_xlabel(f"Dim {d1+1}")
    ax[0].set_ylabel(f"Dim {d2+1}")
    ax[0].legend(loc='upper right')

    c1 = ax[1].contourf(G1, G2, ei_2d, cmap='plasma')
    fig.colorbar(c1, ax=ax[1])
    ax[1].scatter(max_x1, max_x2, color='black', s=80, marker='*', label='EI slice max')
    ax[1].set_title(f"EI (dims {d1+1} vs {d2+1})")
    ax[1].set_xlabel(f"Dim {d1+1}")
    ax[1].set_ylabel(f"Dim {d2+1}")
    ax[1].legend(loc='upper right')

    plt.tight_layout()
    plt.show()

def _format_for_portal(x):
    # Keep within [0, 0.999999] so all values begin with '0' and have 6 dp.
    # (Capstone portal requirement & example formatting.)  # noqa
    x = np.clip(np.asarray(x, float), 0.0, 1.0 - 1e-6)
    return "-".join(f"{v:.6f}" for v in x)  # e.g. 0.123456-0.654321  # noq


def _lhs(n, d, rng):
    # Simple Latin Hypercube Sampling in [0,1]^d
    # Avoids clustering vs. pure random.
    cut = np.linspace(0, 1, n + 1)
    a = cut[:-1]
    b = cut[1:]
    X = rng.uniform(a, b, size=(d, n)).T  # shape (n, d)
    for j in range(d):
        rng.shuffle(X[:, j])
    return X




def propose_precise_next_queries(
    X, Y, gp,
    n_global=4000,
    n_trust=32,
    trust_radius=0.03,
    k_final=5,
    random_state=0
):
    """
    Global EI search (LHS) + local trust-region around X_best.
    Returns (numeric_list, formatted_list) de-duplicated to 6 dp.
    """
    rng = np.random.default_rng(random_state)
    d = X.shape[1]

    # Use GP training scale for EI consistency
    Y_best = float(np.max(gp.y_train_))
    # Trust-region center around the best training point used by the GP
    X_best = gp.X_train_[int(np.argmax(gp.y_train_))]


    # (1) Global candidates via LHS in [0,1]^d
    Xg = _lhs(n_global, d, rng)

    # (2) Trust-region candidates around current best
    deltas = rng.uniform(-trust_radius, trust_radius, size=(n_trust, d))
    Xt = np.clip(X_best + deltas, 0.0, 1.0 - 1e-6)

    # Combine & score
    Xc = np.vstack([Xg, Xt])
    ei = expected_improvement(Xc, gp, Y_best)
    ucb = upper_confidence_bound(Xc, gp, kappa=2.0)

    # Order by EI then UCB (tie-break)
    order = np.lexsort((-ucb, -ei))
    Xc_sorted = Xc[order]

    # De-duplicate by 6 dp (portal formatting)
    seen, numeric, formatted = set(), [], []
    for x in Xc_sorted:
        fx = _format_for_portal(x)
        if fx not in seen:
            seen.add(fx)
            numeric.append(x)
            formatted.append(fx)
            if len(numeric) >= k_final:
                break
    return numeric, formatted



def generate_hybrid_query(function_path, n_global=2000, n_trust=24, k_final=1, random_state=0):
    """
    Loads latest data for the given function folder (e.g., 'function_1'),
    fits GP, and returns (numeric_query, formatted_query).
    """
    X = np.load(f'{function_path}/new_inputs.npy')
    Y = np.load(f'{function_path}/new_outputs.npy')
    if X.ndim == 1:  # ensure 2D
        X = X.reshape(-1, 1)

    gp, _ = fit_gp(X, Y, random_state=random_state)
    numeric_list, formatted_list = propose_precise_next_queries(
        X, Y, gp,
        n_global=n_global,
        n_trust=n_trust,
        k_final=max(1, k_final),
        random_state=random_state
    )
    return numeric_list[0], formatted_list[0]

def run_function_workflow(function_path, random_state=0):
    # 1) Load

        X = X.reshape(-1, 1)

    gp, _ = fit_gp(X, Y, random_state=random_state)

    # 3) Report current best
    idx_best = int(np.argmax(Y))
